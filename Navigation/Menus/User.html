def initialize(user)
  @key           = :user
  @display       = Katello.config[:gravatar] ? "#{gravatar_image_tag(user.email)}#{user.username}" : user.username
  @authorization = true
  @type          = 'dropdown'
  @items         = [
    Navigation::Items::UserAccount.new(user),
    Navigation::Items::Logout.new
  ]
  super
end
