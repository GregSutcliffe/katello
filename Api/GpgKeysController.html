def rules
  read_test = lambda{@gpg_key.readable?}
  manage_test = lambda{@gpg_key.manageable?}
  create_test = lambda{GpgKey.createable?(@organization)}
  index_test = lambda{GpgKey.any_readable?(@organization)}
  {
    :index => index_test,
    :show => read_test,
    :create => create_test,
    :update => manage_test,
    :destroy => manage_test
  }
end
