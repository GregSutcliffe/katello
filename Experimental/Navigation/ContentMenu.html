def menu_repositories
  {
    :key    => :providers,
    :display=> _("Repositories"),
    :url    => providers_path,
    :type   => 'flyout',
    :items  => [
      {
        :key    => :redhat_providers,
        :display=> _("Red Hat Repositories"),
        :url    => redhat_provider_providers_path,

      },{
        :key    => :custom_providers,

        :display=> _("GPG Keys"),
        :url    => gpg_keys_path,
        :if     => lambda{ GpgKey.any_readable?(@current_organization) },
     }
    ]
  }
